#KFold5
#k_fold5c_best
def k_fold(X_train,y_train,params_c,lr_penalty):
    #KFold5
    kf = KFold(n_splits=5, random_state=2019, shuffle=True)
    #cf1_score
    c_best = params_c[0]
    best_f1Score = 0
    #params_c
    for c in params_c:
        curr_f1Score = 0
        #f1_score
        for train_index, valid_index in kf.split(X_train): 
            #l1Cc
            LR = LogisticRegression(C=c,penalty=lr_penalty,
                                    solver='liblinear',class_weight='balanced')
            #
            LR.fit(X_train[train_index], y_train[train_index])
            #f1_score
            curr_f1Score += f1_score(y_train[valid_index],LR.predict(X_train[valid_index]))
        #5f1_score
        avg_f1Score = curr_f1Score/5
        #cf1_score
        if avg_f1Score > best_f1Score:
            c_best = c
            best_f1Score = avg_f1Score
        #
        print ("f1_score: %.2f"%best_f1Score, "c:{}".format(c_best))
    #c_best
    print ("c:{}".format(c_best))
